# Now Create the Mysql binlog directory ( otherwise deafults to /var/lib/mysql)

[root@instance-00000047 mysql]# pwd
/var/lib/mysql
[root@instance-00000047 mysql]# mkdir binary_log
[root@instance-00000047 mysql]# chown mysql:mysql binary_log

Now create the /etc/.my.cnf file ( mysql-libs rpm installs this file )

[mysqld]
server-id = 1
log-bin = /var/lib/mysql/binary_log/mysql-bin.log
binlog-do-db = sojourner
binlog-ignore-db = mysql
binlog_format=mixed

[mysqld_safe]
log-error=/var/log/mysqld.log

On Client 



Create Replication user on Server
GRANT REPLICATION SLAVE ON *.* TO 'repl'@'10.0.0.%' IDENTIFIED BY 'replication';
FLUSH PRIVILEGES;

Confirm from slave that repl user can connect
